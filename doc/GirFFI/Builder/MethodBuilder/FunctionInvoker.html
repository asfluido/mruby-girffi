def invoke_function sym,*o,&b
  p [:ivoked, sym, *o, b] if GirFFI::DEBUG[:VERBOSE]

  o = o.map do |q|
    if q.respond_to?(:to_ptr)
      next q.to_ptr
    end
    
    if q.is_a?(::String)
      next "#{q}"
    end
    
    if q == nil
      next FFI::Pointer::NULL
    end
    
    q
  end
  
  return send sym,*o,&b    
end
